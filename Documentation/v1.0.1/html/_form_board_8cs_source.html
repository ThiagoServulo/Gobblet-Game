106 private void ButtonStartClick(object sender, EventArgs e)
107 {
108 try
109 {
110 int roundQuantity = int.Parse(this.textBoxRoundQuantity.Text);
111
112 if (((roundQuantity % 2) == 0) || (roundQuantity < 0))
113 {
114 throw new FormatException();
115 }
116
117 this.textBoxRoundQuantity.Enabled = false;
118 this.buttonStart.Enabled = false;
119 this.StartGame(roundQuantity);
120 }
121 catch(FormatException)
122 {
123 MessageBox.Show("Error: The number of rounds must be a positive odd integer.");
124 this.InitFormBoard();
