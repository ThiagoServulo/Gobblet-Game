43 public PlayStatus Play(Field field, Player player)
44 {
45 // If the source field is empty, it will be assigned
46 if (this.Origin == null)
47 {
48 // Check if the source field is valid for that player
49 if (!this.IsValidOrigin(field, player))
50 {
51 return PlayStatus.WaitOriginField;
52 }
53 else
54 {
55 this.Origin = field;
56 this.Origin.ChangeCircleColor();
57 return PlayStatus.WaitDestinyField;
58 }
59 }
60
61 // If the field is selected twice, it will no longer be considered as the source
62 if (this.Origin == field)
63 {
64 this.Origin.ChangeCircleColor();
65 this.Origin = null;
66 return PlayStatus.WaitOriginField;
67 }
68
69 // If another source field is selected, the old field returns to its initial
70 // state and the new one is assigned
71 if ((field.FieldPictureBox.Name.Contains('X') && (player.Type == PlayerType.PlayerX)) ||
72 (field.FieldPictureBox.Name.Contains('Y') && (player.Type == PlayerType.PlayerY)))
73 {
74 this.Origin.ChangeCircleColor();
75 this.Origin = field;
76 this.Origin.ChangeCircleColor();
77 return PlayStatus.WaitDestinyField;
78 }
79
80 // Check if the destination field is valid for that player
81 if (this.IsValidDestiny(field))
82 {
83 this.Destiny = field;
84 this.MovePiece();
85 return PlayStatus.Finish;
86 }
87
88 // If none of the above conditions are met, the selected field is assigned as the destination
89 return PlayStatus.WaitDestinyField;
90 }

121 public bool IsValidOrigin(Field field, Player player)
122 {
123 // A player can only select initial fields corresponding to their color
124 if ((field.FieldPictureBox.Name.Contains('X') && (player.Type == PlayerType.PlayerX)) ||
125 (field.FieldPictureBox.Name.Contains('Y') && (player.Type == PlayerType.PlayerY)))
126 {
127 return true;
128 }
129
130 // A player cannot select an initial field that does not contain pieces
131 if (field.GetLast() == null)
132 {
133 return false;
134 }
135
136 // A player can only select an initial field that contains a piece of their color
137 if (((field.GetLast().Color.Primary == Color.DarkRed) && (player.Type == PlayerType.PlayerX)) ||
138 ((field.GetLast().Color.Primary == Color.DarkBlue) && (player.Type == PlayerType.PlayerY)))
139 {
140 return true;
141 }
142
143 // If none of the above conditions are met, the source field is invalid for the move
144 return false;
145 }

156 public bool IsValidDestiny(Field field)
157 {
158 // The destination field cannot be an initial field
159 if (field.FieldPictureBox.Name.Contains('X') || field.FieldPictureBox.Name.Contains('Y'))
160 {
161 return false;
162 }
163
164 // The destination field can be an empty field
165 if (field.GetLast() == null)
166 {
167 return true;
168 }
169
170 // The destination field will only be valid if the piece contained in the
171 // source field is larger than the piece present in the destination field
172 if (this.Origin.GetLast().CompareTo(field.GetLast()) > 0)
173 {
174 return true;
175 }
176
177 // If none of the above conditions are met, the destination field is
178 // invalid for the move
179 return false;
180 }
