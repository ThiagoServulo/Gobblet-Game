84 public void InitBoard()
85 {
86 // Movement initialization
87 this.PlayerMove = new Move();
88
89 int indexFieldsPlayerX = 0;
90 int indexFieldsPlayerY = 0;
91 int indexFieldsBoard = 0;
92
93 for (int i = 0; i < this.PictureBoxes.Length; i++)
94 {
95 if (this.PictureBoxes[i].Name.Contains('X'))
96 {
97 // Grouping of initial fields for Player X
98 this.InitFieldsPlayerX[indexFieldsPlayerX] = new Field(PictureBoxes[i]);
99 indexFieldsPlayerX += 1;
100 }
101 else if (this.PictureBoxes[i].Name.Contains('Y'))
102 {
103 // Grouping of initial fields for Player Y
104 this.InitFieldsPlayerY[indexFieldsPlayerY] = new Field(PictureBoxes[i]);
105 ++indexFieldsPlayerY;
106 }
107 else
108 {
109 // Grouping of fields on the board common to both players
110 this.FieldsBoard[indexFieldsBoard] = new Field(PictureBoxes[i]);
111 ++indexFieldsBoard;
112 }
113 }
114
115 // Initialization of pieces for Player X
116 this.InitFields(this.InitFieldsPlayerX, Color.DarkRed, Color.IndianRed);
117
118 // Initialization of pieces for Player Y
119 this.InitFields(this.InitFieldsPlayerY, Color.DarkBlue, Color.LightBlue);
120
121 // Board initialization
122 this.InitFields(this.FieldsBoard, null, null);
123
124 // Concatenation of all fields on the board
125 this.AllFields = new Field[][] { this.FieldsBoard, this.InitFieldsPlayerY, this.InitFieldsPlayerX };
126 }

137 public void InitFields(Field[] fields, Color? primaryColor, Color? secundaryColor)
138 {
139 for (int i = 0; i < fields.Length; i++)
140 {
141 // Enable the field to be selected
142 fields[i].FieldPictureBox.Enabled = true;
143
144 if (primaryColor == null || secundaryColor == null) // If the colors are null, the field will be cleared
145 {
146 fields[i].DrawCircle(null);
147 }
148 else if (i < 3) // Add the smallest existing circle
149 {
150 fields[i].AddCircle(new Circle((Color)primaryColor, (Color)secundaryColor, CircleType.Type1));
151 }
152 else if (i < 6) // Add the second smallest existing circle
153 {
154 fields[i].AddCircle(new Circle((Color)primaryColor, (Color)secundaryColor, CircleType.Type2));
155 }
156 else if (i < 9) // Add the second largest existing circle
157 {
158 fields[i].AddCircle(new Circle((Color)primaryColor, (Color)secundaryColor, CircleType.Type3));
159 }
160 else // Add the largest existing circle
161 {
162 fields[i].AddCircle(new Circle((Color)primaryColor, (Color)secundaryColor, CircleType.Type4));
163 }

175 public bool Click(PictureBox pictureBox)
176 {
177 Field field = this.GetField(pictureBox);
178
179 // If the move has been completed, it will switch to the next player
180 if (this.PlayerMove.Play(field, this.CurrentPlayer) == PlayStatus.Finish)
181 {
182 Player playerWinner = this.CheckWinner();
183 if (playerWinner != null)
184 {
185 playerWinner.AddVictory();
186 string namePlayer = this.UpdateLabelPlayer(playerWinner);
187 this.ShowMessageBoxWinner(namePlayer);
188 this.InitBoard();
189 this.CurrentRound += 1;
190 if (this.CurrentRound > this.RoundQuantity)
191 {
192 if (MessageBox.Show($"Victory of {namePlayer}!\nDo you want to play another match?",
193 "End of game!", MessageBoxButtons.YesNo) == DialogResult.No)
194 {
195 // Terminate the application.
196 Application.Exit();
197 }
198 return true;
199 }
200 }
201 this.ChangeCurrentPlayer();
202 }
203
204 return false;
205 }

283 public Player CheckWinner()
284 {
285 // Check columns
286 for(int i = 0; i <= 12; i+=4)
287 {
288 if (this.FieldsBoard[i].Equals(this.FieldsBoard[i + 1]) &&
289 this.FieldsBoard[i].Equals(this.FieldsBoard[i + 2]) &&
290 this.FieldsBoard[i].Equals(this.FieldsBoard[i + 3]))
291 {
292 return this.CheckPlayer(this.FieldsBoard[i]);
293 }
294 }
295
296 // Check rows
297 for (int i = 0; i <= 4; i++)
298 {
299 if (this.FieldsBoard[i].Equals(this.FieldsBoard[i + 4]) &&
300 this.FieldsBoard[i].Equals(this.FieldsBoard[i + 8]) &&
301 this.FieldsBoard[i].Equals(this.FieldsBoard[i + 12]))
302 {
303 return this.CheckPlayer(this.FieldsBoard[i]);
304 }
305 }
306
307 // Check main diagonal
308 if (this.FieldsBoard[0].Equals(this.FieldsBoard[5]) &&
309 this.FieldsBoard[0].Equals(this.FieldsBoard[10]) &&
310 this.FieldsBoard[0].Equals(this.FieldsBoard[15]))
311 {
312 return this.CheckPlayer(this.FieldsBoard[0]);
313 }
314
315 // Check secondary diagonal
316 if (this.FieldsBoard[3].Equals(this.FieldsBoard[6]) &&
317 this.FieldsBoard[3].Equals(this.FieldsBoard[9]) &&
318 this.FieldsBoard[3].Equals(this.FieldsBoard[12]))
319 {
320 return this.CheckPlayer(this.FieldsBoard[3]);
321 }
322
323 // Returns null if there is no winner
324 return null;
325 }

336 public Player CheckPlayer(Field field)

344 return this.PlayerX;
